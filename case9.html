1.C Program to Find Factorial of a Number.
2.The factorial of a positive integer n is equal to 1*2*3*...5. You will learn to calculate the factorial of a number using for loop .

For Example-:

	#include<stdio.h>
	int factorial(int);
	void main()
	{
	   int i,n,fact=0;
	   printf("enter the no");
	   scanf("%d",&n);
	   fact=factorial(n);
	   printf("factorial value:%d",fact);
	   
	}
	int factorial(int m);
	{
	   int i,fact=1;
	   if(m==0)
	   {
	   return(0);
	   }
	   else
	   {
	    for(i=1;i<=m;i++)
		{
		   fact=fact*i;
		}
		 return(fact);
	    }
	}

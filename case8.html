1.Create star triangle pattern in C by using nested for loop.

For Example-:

	#include<stdio.h>
	void main()
	{
	   int i,j;
	   for(i=0;i<5;i++)
	   
	   {
	     for(j=0;j<5;j++)
		{
		 if(j<=i)
		 {
		 printf("*");
		 }
		 else
		 {
		 printf(" ");
		 }
		 printf("\n");
		
		}
	   }
	 }
